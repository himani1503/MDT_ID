        // Page navigation
        function showPage(pageId) {
            // Hide all pages
            document.querySelectorAll('.page').forEach(page => {
                page.classList.remove('active');
                page.style.display = 'none';
            });
            
            // Show selected page
            const targetPage = document.getElementById(pageId);
            if (targetPage) {
                targetPage.classList.add('active');
                targetPage.style.display = 'block';
            }
        }

        // Tab switching
        function showTab(tabId) {
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('active');
            });
            document.querySelectorAll('.form-section').forEach(section => {
                section.classList.remove('active');
            });
            
            event.target.classList.add('active');
            document.getElementById(tabId).classList.add('active');
        }

        // Scroll to features
        function scrollToFeatures() {
            document.getElementById('features').scrollIntoView({ behavior: 'smooth' });
        }

        // Generate MedTrack ID
        function generateMedTrackId() {
            const year = new Date().getFullYear();
            const randomNum = Math.floor(100000 + Math.random() * 900000);
            return `MT-${year}-${randomNum}`;
        }

        // Simple function to go directly to dashboard
        function goToDashboard() {
            // Set default user
            currentUser = users['MT-2024-001234'];
            
            // Show dashboard page
            showPage('dashboardPage');
            
            // Load dashboard data
            loadDashboard();
        }

        // Load dashboard data
        function loadDashboard() {
            if (!currentUser) return;
            
            // Update user info
            document.getElementById('userName').textContent = currentUser.name;
            document.getElementById('userMedId').textContent = currentUser.id;
            document.getElementById('userAvatar').textContent = currentUser.name.split(' ').map(n => n[0]).join('').toUpperCase();
            
            // Load purchase history
            loadPurchaseHistory();
            
            // Check for alerts
            checkForAlerts();
        }

        // Load purchase history
        function loadPurchaseHistory() {
            const historyContainer = document.getElementById('purchaseHistory');
            historyContainer.innerHTML = '';
            
            purchaseHistory.forEach(purchase => {
                const historyItem = document.createElement('div');
                historyItem.className = 'history-item';
                
                const typeTag = purchase.type === 'Prescription' ? 
                    '<span class="prescription-tag">Prescription</span>' : 
                    '<span class="otc-tag">OTC</span>';
                
                const doctorInfo = purchase.doctor ? 
                    `<br><small>Prescribed by: ${purchase.doctor}</small>` : '';
                
                historyItem.innerHTML = `
                    <div style="display: flex; justify-content: space-between; align-items: center;">
                        <div>
                            <strong>${purchase.medicine}</strong> ${typeTag}
                            <br><small>Quantity: ${purchase.quantity} | Store: ${purchase.store}</small>
                            ${doctorInfo}
                        </div>
                        <div style="text-align: right;">
                            <small>${new Date(purchase.date).toLocaleDateString()}</small>
                        </div>
                    </div>
                `;
                
                historyContainer.appendChild(historyItem);
            });
        }

        // Check for alerts
        function checkForAlerts() {
            const prescriptionMeds = purchaseHistory.filter(p => p.type === 'Prescription').length;
            const recentNarcotics = purchaseHistory.filter(p => 
                p.medicine.toLowerCase().includes('tramadol') || 
                p.medicine.toLowerCase().includes('codeine') ||
                p.medicine.toLowerCase().includes('morphine')
            ).length;
            
            if (recentNarcotics > 0) {
                document.getElementById('alertSection').style.display = 'block';
                document.getElementById('alertMessage').textContent = 
                    'Narcotic purchases detected. This activity is being monitored for safety compliance.';
            }
        }
